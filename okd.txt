- criar service 

curl -X POST \
     -H "Authorization: Bearer $TOKEN" \
     -H "Content-Type: application/json" \
     -k \
     https://api.cluster.example.com:6443/api/v1/namespaces/NOME_DO_NAMESPACE/services \
     -d '{
       "apiVersion": "v1",
       "kind": "Service",
       "metadata": {
         "name": "NOME_DO_SERVICO",
         "namespace": "NOME_DO_NAMESPACE"
       },
       "spec": {
         "selector": {
           "app": "NOME_DO_LABEL_APP"
         },
         "ports": [
           {
             "protocol": "TCP",
             "port": NOME_DA_PORTA_NO_SERVICO,
             "targetPort": NOME_DA_PORTA_NO_CONTAINER
           }
         ]
       }
     }'

- criar rota

	 curl -X POST \
     -H "Authorization: Bearer $TOKEN" \
     -H "Content-Type: application/json" \
     -k \
     https://api.cluster.example.com:6443/apis/route.openshift.io/v1/namespaces/NOME_DO_NAMESPACE/routes \
     -d '{
       "apiVersion": "route.openshift.io/v1",
       "kind": "Route",
       "metadata": {
         "name": "NOME_DA_ROTA",
         "namespace": "NOME_DO_NAMESPACE"
       },
       "spec": {
         "to": {
           "kind": "Service",
           "name": "NOME_DO_SERVICO"
         },
         "port": {
           "targetPort": "NOME_DA_PORTA_NO_SERVICO"
         },
         "wildcardPolicy": "Subdomain"
       }
     }'

     -d '{
       "spec": {
